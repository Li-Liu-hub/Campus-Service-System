# 私信聊天模块

## 模块简介

私信聊天模块提供用户之间的一对一实时聊天功能，基于WebSocket实现。

## 功能清单

### 1. 私信发送
- 一对一聊天
- 文本消息
- 实时送达

### 2. 会话管理
- 会话列表
- 最近联系人
- 未读消息数

### 3. 消息历史
- 历史消息查询
- 消息分页加载

### 聊天 Service
```java
/**
 * 会话服务实现
 */
@Service
public class ConversationServiceImpl implements ConversationService {

    @Autowired
    private ConversationMapper conversationMapper;

    @Autowired
    private PrivateMessageMapper privateMessageMapper;

    @Autowired
    private ChatWebSocketHandler webSocketHandler;

    /**
     * 发送私信
     */
    @Override
    public void sendMessage(Integer senderId, Integer receiverId, String content) {
        // 1. 获取或创建会话
        Long conversationId = getOrCreateConversation(senderId, receiverId);

        // 2. 保存消息
        PrivateMessage message = PrivateMessage.builder()
                .conversationId(conversationId)
                .senderId(senderId)
                .receiverId(receiverId)
                .content(content)
                .messageType(1)  // 文本消息
                .isRead(0)
                .createTime(LocalDateTime.now())
                .build();
        privateMessageMapper.insert(message);

        // 3. 更新会话最后消息
        conversationMapper.updateLastMessage(conversationId, content, LocalDateTime.now());

        // 4. 增加未读数
        conversationMapper.incrementUnreadCount(conversationId, receiverId);

        // 5. WebSocket 实时推送
        try {
            WebSocketMessageDTO wsMessage = WebSocketMessageDTO.builder()
                    .type("chat")
                    .data(message)
                    .build();
            webSocketHandler.sendMessageToUser(receiverId.longValue(), wsMessage);
        } catch (IOException e) {
            log.error("WebSocket推送失败", e);
        }
    }

    /**
     * 获取或创建会话
     */
    private Long getOrCreateConversation(Integer userId1, Integer userId2) {
        // 确保 userId1 < userId2，保持一致性
        Integer minId = Math.min(userId1, userId2);
        Integer maxId = Math.max(userId1, userId2);

        Conversation conversation = conversationMapper.getByUsers(minId, maxId);
        if (conversation != null) {
            return conversation.getId();
        }

        // 创建新会话
        conversation = Conversation.builder()
                .userId1(minId)
                .userId2(maxId)
                .unreadCount1(0)
                .unreadCount2(0)
                .build();
        conversationMapper.insert(conversation);
        return conversation.getId();
    }

    /**
     * 获取会话列表
     */
    @Override
    public List<Conversation> getConversationList(Integer userId) {
        return conversationMapper.getByUserId(userId);
    }

    /**
     * 获取会话消息历史
     */
    @Override
    public List<PrivateMessage> getMessages(Long conversationId, Integer page, Integer pageSize) {
        return privateMessageMapper.getByConversationId(
                conversationId, (page - 1) * pageSize, pageSize);
    }

    /**
     * 标记消息已读
     */
    @Override
    public void markAsRead(Long conversationId, Integer userId) {
        // 标记消息为已读
        privateMessageMapper.markAsRead(conversationId, userId);

        // 重置未读数
        conversationMapper.resetUnreadCount(conversationId, userId);
    }
}
```

## 数据库表

### conversation（会话表）
```sql
CREATE TABLE `conversation` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '会话ID',
  `user_id_1` int NOT NULL COMMENT '用户1ID',
  `user_id_2` int NOT NULL COMMENT '用户2ID',
  `last_message` text COMMENT '最后一条消息',
  `last_message_time` datetime DEFAULT NULL COMMENT '最后消息时间',
  `unread_count_1` int DEFAULT 0 COMMENT '用户1未读数',
  `unread_count_2` int DEFAULT 0 COMMENT '用户2未读数',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_users` (`user_id_1`, `user_id_2`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='会话表';
```

### private_message（私信表）
```sql
CREATE TABLE `private_message` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '消息ID',
  `conversation_id` bigint NOT NULL COMMENT '会话ID',
  `sender_id` int NOT NULL COMMENT '发送者ID',
  `receiver_id` int NOT NULL COMMENT '接收者ID',
  `content` text NOT NULL COMMENT '消息内容',
  `message_type` tinyint DEFAULT 1 COMMENT '消息类型: 1-文本 2-图片',
  `is_read` tinyint DEFAULT 0 COMMENT '是否已读: 0-未读 1-已读',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  INDEX `idx_conversation_id`(`conversation_id`),
  INDEX `idx_sender_id`(`sender_id`),
  INDEX `idx_receiver_id`(`receiver_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='私信消息表';
```

## 前端实现

### WebSocket 工具类
```typescript
// vue3-front/src/utils/websocket.ts

import { useUserStore } from '@/stores/user';

class WebSocketService {
  private ws: WebSocket | null = null;
  private reconnectTimer: number | null = null;
  private messageHandlers: Map<string, Function[]> = new Map();

  // 连接 WebSocket
  connect() {
    const userStore = useUserStore();
    const userId = userStore.userInfo?.id;

    if (!userId) return;

    // 创建 WebSocket 连接
    this.ws = new WebSocket(`ws://localhost:8080/ws/chat?userId=${userId}`);

    this.ws.onopen = () => {
      console.log('WebSocket 连接成功');
    };

    this.ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      this.handleMessage(message);
    };

    this.ws.onclose = () => {
      console.log('WebSocket 连接关闭');
      this.reconnect();
    };
  }

  // 处理接收到的消息
  private handleMessage(message: any) {
    const handlers = this.messageHandlers.get(message.type);
    if (handlers) {
      handlers.forEach(handler => handler(message.data));
    }
  }

  // 订阅消息
  on(type: string, handler: Function) {
    if (!this.messageHandlers.has(type)) {
      this.messageHandlers.set(type, []);
    }
    this.messageHandlers.get(type)!.push(handler);
  }

  // 发送消息
  send(message: any) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(message));
    }
  }

  // 断开连接
  disconnect() {
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
    }
    this.ws?.close();
  }
}

export const wsService = new WebSocketService();
```
