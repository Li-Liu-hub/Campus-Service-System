# 秒杀模块

## 模块简介

秒杀模块提供限时抢购功能，包括秒杀商品和秒杀订单两种形式。

## 功能清单

### 1. 秒杀商品
- 秒杀商品列表
- 秒杀时间场次
- 限量抢购
- 秒杀订单生成

### 2. 秒杀订单
- 订单秒杀标记
- 置顶展示
- 额外费用

### 3. 秒杀管理（管理员）
- 添加秒杀商品
- 设置秒杀价格
- 设置秒杀数量
- 开启/关闭秒杀

### 秒杀 Service
```java
/**
 * 秒杀服务实现
 * 使用 Redis + RabbitMQ 实现高并发秒杀
 */
@Service
public class SeckillServiceImpl implements SeckillService {

    @Autowired
    private StringRedisTemplate redisTemplate;

    @Autowired
    private RabbitTemplate rabbitTemplate;

    @Autowired
    private SeckillGoodsMapper seckillGoodsMapper;

    @Autowired
    private SeckillOrderMapper seckillOrderMapper;

    private static final String SECKILL_STOCK_KEY = "seckill:stock:";
    private static final String SECKILL_ORDER_KEY = "seckill:order:";

    /**
     * 秒杀请求
     * @param userId 用户ID
     * @param goodsId 商品ID
     */
    @Override
    public void seckill(Long userId, Long goodsId) {
        // 1. 检查秒杀时间
        SeckillGoods goods = seckillGoodsMapper.getById(goodsId);
        if (goods == null) {
            throw new SeckillException("商品不存在");
        }

        LocalDateTime now = LocalDateTime.now();
        if (now.isBefore(goods.getStartTime())) {
            throw new SeckillException("秒杀还未开始");
        }
        if (now.isAfter(goods.getEndTime())) {
            throw new SeckillException("秒杀已结束");
        }

        // 2. 检查库存（从 Redis 获取，高性能）
        String stockKey = SECKILL_STOCK_KEY + goodsId;
        Integer stock = redisTemplate.opsForValue().get(stockKey, Integer.class);
        if (stock == null) {
            // 初始化库存到 Redis
            stock = goods.getSeckillStock() - goods.getSeckillCount();
            redisTemplate.opsForValue().set(stockKey, stock.toString());
        }

        if (stock <= 0) {
            throw new SeckillException("商品已抢光");
        }

        // 3. 检查是否已秒杀（防止重复秒杀）
        String orderKey = SECKILL_ORDER_KEY + goodsId + ":" + userId;
        Boolean isOrdered = redisTemplate.hasKey(orderKey);
        if (Boolean.TRUE.equals(isOrdered)) {
            throw new SeckillException("您已抢购过该商品");
        }

        // 4. 减库存（原子操作）
        Long remainingStock = redisTemplate.opsForValue().decrement(stockKey);
        if (remainingStock < 0) {
            // 库存不足，回滚
            redisTemplate.opsForValue().increment(stockKey);
            throw new SeckillException("商品已抢光");
        }

        // 5. 发送消息到 RabbitMQ 异步处理订单
        SeckillMessageDTO message = SeckillMessageDTO.builder()
                .userId(userId)
                .goodsId(goodsId)
                .build();
        rabbitTemplate.convertAndSend("seckill.exchange", "seckill.route", message);
    }

    /**
     * 获取秒杀商品列表
     */
    @Override
    public List<SeckillGoodsVO> getSeckillGoodsList() {
        List<SeckillGoods> goodsList = seckillGoodsMapper.getActiveList();
        return goodsList.stream()
                .map(this::convertToVO)
                .collect(Collectors.toList());
    }

    private SeckillGoodsVO convertToVO(SeckillGoods goods) {
        SeckillGoodsVO vo = new SeckillGoodsVO();
        BeanUtils.copyProperties(goods, vo);

        // 计算状态
        LocalDateTime now = LocalDateTime.now();
        if (now.isBefore(goods.getStartTime())) {
            vo.setStatus(0);  // 未开始
        } else if (now.isAfter(goods.getEndTime())) {
            vo.setStatus(2);  // 已结束
        } else {
            vo.setStatus(1);  // 进行中
        }

        // 计算剩余库存
        int stock = goods.getSeckillStock() - goods.getSeckillCount();
        vo.setStock(stock);

        return vo;
    }
}
```

### 秒杀订单监听器
```java
/**
 * 秒杀订单监听器
 * 监听 RabbitMQ 消息，生成秒杀订单
 */
@Component
public class SeckillOrderListener {

    @Autowired
    private SeckillGoodsMapper seckillGoodsMapper;

    @Autowired
    private SeckillOrderMapper seckillOrderMapper;

    /**
     * 处理秒杀消息
     */
    @RabbitListener(queues = "seckill.queue")
    public void handleSeckillMessage(SeckillMessageDTO message) {
        try {
            // 1. 再次检查库存（数据库层面）
            SeckillGoods goods = seckillGoodsMapper.getById(message.getGoodsId());
            if (goods.getSeckillCount() >= goods.getSeckillStock()) {
                return;  // 库存不足，跳过
            }

            // 2. 扣减库存
            seckillGoodsMapper.incrementSeckillCount(message.getGoodsId());

            // 3. 创建秒杀订单
            SeckillOrder order = SeckillOrder.builder()
                    .userId(message.getUserId().intValue())
                    .goodsId(message.getGoodsId())
                    .orderNo("SK" + System.currentTimeMillis() + UUID.randomUUID().toString().substring(0, 8))
                    .payAmount(goods.getSeckillPrice())
                    .status(0)  // 待支付
                    .createTime(LocalDateTime.now())
                    .build();
            seckillOrderMapper.insert(order);

        } catch (Exception e) {
            log.error("处理秒杀订单失败", e);
            // 可发送失败通知等
        }
    }
}
```

## 数据库表

### seckill_goods（秒杀商品表）
```sql
CREATE TABLE `seckill_goods` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '商品ID',
  `goods_name` varchar(100) NOT NULL COMMENT '商品名称',
  `goods_image` varchar(500) DEFAULT NULL COMMENT '商品图片',
  `seckill_price` decimal(10,2) NOT NULL COMMENT '秒杀价格',
  `original_price` decimal(10,2) NOT NULL COMMENT '原价',
  `seckill_stock` int NOT NULL DEFAULT 0 COMMENT '秒杀数量',
  `seckill_count` int NOT NULL DEFAULT 0 COMMENT '已秒杀数量',
  `start_time` datetime NOT NULL COMMENT '秒杀开始时间',
  `end_time` datetime NOT NULL COMMENT '秒杀结束时间',
  `status` tinyint NOT NULL DEFAULT 0 COMMENT '状态: 0-未开始 1-进行中 2-已结束',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  INDEX `idx_status`(`status`),
  INDEX `idx_start_time`(`start_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='秒杀商品表';
```

## 技术架构

```
用户请求
    │
    ▼
┌─────────────────┐
│   限流过滤      │  ← 使用 Redis 计数器限流
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  库存检查       │  ← 从 Redis 获取库存，高性能
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  防重复检查     │  ← 使用 Redis Set 记录已秒杀用户
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│   Redis 扣减库存 │  ← 原子操作，保证线程安全
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  RabbitMQ 消息  │  ← 异步处理订单，削峰填谷
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  订单监听器     │  ← 生成秒杀订单，扣减数据库库存
└─────────────────┘
```
