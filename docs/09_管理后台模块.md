# 管理后台模块

## 模块简介

管理后台模块为管理员提供系统管理功能，包括用户管理、订单管理、帖子管理、数据统计等。

## 功能清单

### 1. 数据统计
- 用户数量统计及趋势
- 订单数量统计及趋势
- 帖子数量统计及趋势
- 交易金额统计及趋势
- ECharts 可视化图表

### 2. 用户管理
- 用户列表查询
- 禁用/启用用户
- 查看用户详情
- 重置用户密码

### 3. 帖子管理
- 帖子列表查询
- 删除违规帖子
- 查看帖子详情

### 4. 订单管理
- 订单列表查询
- 查看订单详情
- 取消订单

### 5. 通知管理
- 发布系统公告
- 查看通知列表
- 删除通知

### 6. 敏感词管理
- 敏感词列表
- 添加敏感词
- 删除敏感词

### 7. 系统设置
- 系统参数配置
- 功能开关

### 8. 操作日志
- 管理员操作记录
- 日志查询

## 技术实现

### 统计数据 Controller
```java
/**
 * 统计控制器
 * 提供各类数据统计接口
 */
@RestController
@RequestMapping("/jsyl/admin/statistics")
public class StatisticsController {

    @Autowired
    private UserMapper userMapper;

    @Autowired
    private OrderCenterMapper orderCenterMapper;

    @Autowired
    private PostMapper postMapper;

    /**
     * 获取用户统计数据
     * @return 用户统计数据
     */
    @GetMapping("/userCount")
    public Result<Long> getUserCount() {
        Long count = userMapper.count();
        return Result.success(count);
    }

    /**
     * 获取订单统计数据
     * @return 订单统计数据
     */
    @GetMapping("/orderCount")
    public Result<Long> getOrderCount() {
        Long count = orderCenterMapper.count();
        return Result.success(count);
    }

    /**
     * 获取帖子统计数据
     * @return 帖子统计数据
     */
    @GetMapping("/postCount")
    public Result<Long> getPostCount() {
        Long count = postMapper.count();
        return Result.success(count);
    }

    /**
     * 获取交易金额统计
     * @return 交易总金额
     */
    @GetMapping("/totalAmount")
    public Result<Double> getTotalAmount() {
        Double total = orderCenterMapper.sumAmount();
        return Result.success(total);
    }

    /**
     * 获取用户增长趋势（最近N天）
     * @param days 天数
     * @return 每日新增用户数
     */
    @GetMapping("/userTrend")
    public Result<List<Map<String, Object>>> getUserTrend(@RequestParam(defaultValue = "30") Integer days) {
        LocalDateTime startDate = LocalDateTime.now().minusDays(days);
        List<Map<String, Object>> trend = userMapper.getUserTrend(startDate);
        return Result.success(trend);
    }

    /**
     * 获取订单增长趋势
     * @param days 天数
     * @return 每日新增订单数
     */
    @GetMapping("/orderTrend")
    public Result<List<Map<String, Object>>> getOrderTrend(@RequestParam(defaultValue = "30") Integer days) {
        LocalDateTime startDate = LocalDateTime.now().minusDays(days);
        List<Map<String, Object>> trend = orderCenterMapper.getOrderTrend(startDate);
        return Result.success(trend);
    }

    /**
     * 获取帖子增长趋势
     * @param days 天数
     * @return 每日新增帖子数
     */
    @GetMapping("/postTrend")
    public Result<List<Map<String, Object>>> getPostTrend(@RequestParam(defaultValue = "30") Integer days) {
        LocalDateTime startDate = LocalDateTime.now().minusDays(days);
        List<Map<String, Object>> trend = postMapper.getPostTrend(startDate);
        return Result.success(trend);
    }
}
```

### 用户管理 Controller
```java
/**
 * 管理员用户管理控制器
 */
@RestController
@RequestMapping("/jsyl/admin/user")
public class AdminUserController {

    @Autowired
    private UserService userService;

    /**
     * 分页查询用户列表
     */
    @GetMapping("/pageQuery")
    public Result<PageResult> pageQuery(UserPageQueryDTO dto) {
        PageResult pageResult = userService.adminPageQuery(dto);
        return Result.success(pageResult);
    }

    /**
     * 禁用/启用用户
     */
    @PutMapping("/status/{id}")
    public Result<String> updateStatus(@PathVariable Long id, @RequestParam Integer status) {
        userService.updateStatus(id, status);
        return Result.success(status == 0 ? "禁用成功" : "启用成功");
    }

    /**
     * 重置用户密码
     */
    @PutMapping("/resetPassword/{id}")
    public Result<String> resetPassword(@PathVariable Long id) {
        userService.resetPassword(id);
        return Result.success("密码已重置为 123456");
    }
}
```

### 操作日志 Aspect
```java
/**
 * 操作日志切面
 * 自动记录Controller方法的操作日志
 */
@Aspect
@Component
public class OperationLogAspect {

    @Autowired
    private OperationLogMapper operationLogMapper;

    /**
     * 切入点：所有带 @OperationLog 注解的方法
     */
    @Pointcut("@annotation(com.jsyl.annotation.OperationLog)")
    public void logPointcut() {}

    /**
     * 环绕通知：记录操作日志
     */
    @Around("logPointcut()")
    public Object around(ProceedingJoinPoint joinPoint) throws Throwable {
        // 1. 获取方法参数
        Object[] args = joinPoint.getArgs();

        // 2. 获取方法上的注解
        MethodSignature signature = (MethodSignature) joinPoint.getSignature();
        OperationLog annotation = signature.getMethod().getAnnotation(OperationLog.class);

        // 3. 记录操作日志
        Long startTime = System.currentTimeMillis();
        String operation = annotation.value();

        try {
            Object result = joinPoint.proceed();
            long duration = System.currentTimeMillis() - startTime;

            // 保存成功日志
            saveLog(operation, "成功", duration);

            return result;
        } catch (Exception e) {
            long duration = System.currentTimeMillis() - startTime;

            // 保存失败日志
            saveLog(operation, "失败: " + e.getMessage(), duration);

            throw e;
        }
    }

    private void saveLog(String operation, String result, long duration) {
        try {
            Long userId = BaseContext.getCurrentId();

            OperationLog log = OperationLog.builder()
                    .userId(userId != null ? userId.intValue() : null)
                    .operation(operation)
                    .result(result)
                    .duration(duration)
                    .createTime(LocalDateTime.now())
                    .build();

            operationLogMapper.insert(log);
        } catch (Exception e) {
            log.error("保存操作日志失败", e);
        }
    }
}
```

## 数据库表

### operation_log（操作日志表）
```sql
CREATE TABLE `operation_log` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '日志ID',
  `user_id` int DEFAULT NULL COMMENT '操作用户ID',
  `operation` varchar(100) NOT NULL COMMENT '操作描述',
  `method` varchar(200) DEFAULT NULL COMMENT '请求方法',
  `params` text COMMENT '请求参数',
  `result` varchar(500) DEFAULT NULL COMMENT '操作结果',
  `duration` bigint DEFAULT NULL COMMENT '耗时(毫秒)',
  `ip` varchar(50) DEFAULT NULL COMMENT 'IP地址',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  INDEX `idx_user_id`(`user_id`),
  INDEX `idx_create_time`(`create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='操作日志表';
```

### system_settings（系统设置表）
```sql
CREATE TABLE `system_settings` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '设置ID',
  `setting_key` varchar(50) NOT NULL COMMENT '设置键',
  `setting_value` varchar(500) DEFAULT NULL COMMENT '设置值',
  `description` varchar(200) DEFAULT NULL COMMENT '说明',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_setting_key` (`setting_key`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统设置表';
```

### sensitive_word（敏感词表）
```sql
CREATE TABLE `sensitive_word` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT 'ID',
  `word` varchar(50) NOT NULL COMMENT '敏感词',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_word` (`word`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='敏感词表';
```
